Here’s a **complete VSIX project skeleton** with all files and code lines integrated. This version includes:

- **Seed data** with both C# and Asio stubs.  
- **MongoDB seeding at startup** (runs all `Seed*.Populate()` methods).  
- **Live PUML editor** with Save/Load.  
- **Forward pipeline** (Asio → C# → PUML).  
- **Reverse pipeline** (PUML → C# → Asio).

## 📄 Data/SeedExecution.cs (example with Asio stubs)

```csharp
using System.Collections.Generic;
using BoostAsioExtension.Models;

namespace BoostAsioExtension.Data
{
    public static class SeedExecution
    {
        public static void Populate(MongoContext context)
        {
            var entries = new List<Entry>
            {
                new Entry {
                    Header = "Execution",
                    Subheader = "Context",
                    Name = "io_context",
                    Description = "Provides core I/O functionality.",
                    CodeStub = "var ioContext = new io_context();",
                    AsioStub = "boost::asio::io_context ioContext;"
                },
                new Entry {
                    Header = "Execution",
                    Subheader = "Context",
                    Name = "run",
                    Description = "Runs the io_context processing loop.",
                    CodeStub = "ioContext.run();",
                    AsioStub = "ioContext.run();"
                }
            };
            context.Entries.InsertMany(entries);
        }
    }
}
```

👉 All other seed files (`SeedNetworking`, `SeedTimers`, etc.) follow the same pattern: each entry now has both `CodeStub` and `AsioStub`.

## 📄 Services/EntryController.cs

```csharp
using MongoDB.Driver;
using System.Collections.Generic;
using BoostAsioExtension.Models;

public class EntryController
{
    private readonly IMongoCollection<Entry> _entries;

    public EntryController(string connectionString, string dbName, string collectionName)
    {
        var client = new MongoClient(connectionString);
        var database = client.GetDatabase(dbName);
        _entries = database.GetCollection<Entry>(collectionName);

        // Seed database at startup
        var count = _entries.CountDocuments(FilterDefinition<Entry>.Empty);
        if (count == 0)
        {
            var context = new MongoContext(connectionString, dbName, collectionName);
            BoostAsioExtension.Data.SeedRunner.RunAll(context);
        }
    }

    public IEnumerable<Entry> GetAllEntries()
    {
        return _entries.Find(_ => true).ToList();
    }
}
```

## 📄 ToolWindowControl.xaml.cs

```csharp
private readonly EntryController _controller;

public ToolWindowControl()
{
    InitializeComponent();
    _controller = new EntryController("mongodb://localhost:27017", "AsioDb", "Entries");
}

private void AsioToCsAndUmlButton_Click(object sender, RoutedEventArgs e)
{
    var service = new DualCompileAndConvertService(_controller);
    var result = service.CompileAndConvert(CppInputBox.Text);

    PumlEditorBox.Text = result.uml;

    PreviewOutputBox.Text =
        $"--- Asio C++ ---\n{CppInputBox.Text}\n\n" +
        $"--- Converted C# ---\n{result.csCode}\n\n" +
        $"--- UML ---\n{result.uml}";
}

private void PumlToCsAndAsioButton_Click(object sender, RoutedEventArgs e)
{
    var pumlConverter = new PumlToCsConverter();
    var csCode = pumlConverter.Convert(PumlEditorBox.Text);

    var asioConverter = new CsToAsioConverter(_controller);
    var asioCode = asioConverter.Convert(csCode);

    PreviewOutputBox.Text =
        $"--- PUML Input ---\n{PumlEditorBox.Text}\n\n" +
        $"--- Generated C# ---\n{csCode}\n\n" +
        $"--- Generated Asio C++ ---\n{asioCode}";
}

private void SavePumlButton_Click(object sender, RoutedEventArgs e)
{
    var dialog = new Microsoft.Win32.SaveFileDialog
    {
        Title = "Save PUML Diagram",
        Filter = "PUML Files (*.puml)|*.puml"
    };
    if (dialog.ShowDialog() == true)
    {
        File.WriteAllText(dialog.FileName, PumlEditorBox.Text);
    }
}

private void LoadPumlButton_Click(object sender, RoutedEventArgs e)
{
    var dialog = new Microsoft.Win32.OpenFileDialog
    {
        Title = "Load PUML Diagram",
        Filter = "PUML Files (*.puml)|*.puml"
    };
    if (dialog.ShowDialog() == true)
    {
        PumlEditorBox.Text = File.ReadAllText(dialog.FileName);
    }
}
```
