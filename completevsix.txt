## 📄 Services/EntryController.cs

```csharp
using BoostAsioExtension.Models;
using MongoDB.Driver;
using System.Collections.Generic;

public class EntryController
{
    private readonly IMongoCollection<Entry> _entries;

    public EntryController(string connectionString, string dbName, string collectionName)
    {
        var client = new MongoClient(connectionString);
        var database = client.GetDatabase(dbName);
        _entries = database.GetCollection<Entry>(collectionName);

        // Seed database at startup if empty
        var count = _entries.CountDocuments(FilterDefinition<Entry>.Empty);
        if (count == 0)
        {
            var context = new MongoContext(connectionString, dbName, collectionName);
            BoostAsioExtension.Data.SeedRunner.RunAll(context);
        }
    }

    public IEnumerable<Entry> GetAllEntries()
    {
        return _entries.Find(_ => true).ToList();
    }
}
```

```csharp
using System;

namespace BoostAsioExtension.Services
{
    public class AsioToCsCompiler
    {
        public string Compile(string asioCode)
        {
            // Simplified: match keywords and return stub
            if (asioCode.Contains("async_read"))
                return "public async Task<int> AsyncRead(byte[] buffer) { return await Task.FromResult(0); }";
            if (asioCode.Contains("async_write"))
                return "public async Task AsyncWrite(byte[] buffer) { await Task.CompletedTask; }";
            return "// Unknown Asio code";
        }
    }
}
```

```csharp
            // Very basic UML generation

            {
                return "@startuml\nclass AsyncRead {\n + int Buffer\n}\n@enduml";
            }
            if (csCode.Contains("AsyncWrite"))
            {
                return "@startuml\nclass AsyncWrite {\n + byte[] Buffer\n}\n@enduml";
            }
            return "@startuml\nclass Unknown {}\n@enduml";
        }
    }
}
```

## 📄 Services/CsToAsioConverter.cs

```csharp
using System.Linq;

namespace BoostAsioExtension.Services
{
    public class CsToAsioConverter
    {
        private readonly EntryController _controller;

        public CsToAsioConverter(EntryController controller)
        {
            _controller = controller;
        }

        public string Convert(string csCode)
        {
            var entries = _controller.GetAllEntries();
            foreach (var entry in entries)
            {
                if (csCode.Contains(entry.Name))
                {
                    return entry.AsioStub;
                }
            }
            return "// Unknown C# code";
        }
    }
}
```

## 📄 Services/DualCompileAndConvertService.cs

```csharp
namespace BoostAsioExtension.Services
{
    public class DualCompileAndConvertService
    {
        private readonly EntryController _controller;

        public DualCompileAndConvertService(EntryController controller)
        {
            _controller = controller;
        }

        public (string csCode, string uml) CompileAndConvert(string asioCode)
        {
            var asioCompiler = new AsioToCsCompiler();
            var csCode = asioCompiler.Compile(asioCode);

            var umlConverter = new CsToUmlConverter();
            var uml = umlConverter.Convert(csCode);

            return (csCode, uml);
        }
    }
}
```

            var asioConverter = new CsToAsioConverter(_controller);

                $"--- Generated C# ---\n{csCode}\n\n" +
                $"--- Generated Asio C++ ---\n{asioCode}";

using Microsoft.VisualStudio.Shell;
using System;
using System.ComponentModel.Design;
using Task = System.Threading.Tasks.Task;

namespace BoostAsioExtension.Commands
{
    internal sealed class BoostAsioCommand
    {
        public const int CommandId = 0x0100;
        public static readonly Guid CommandSet = new Guid("f1a2b3c4-d5e6-7890-abcd-ef1234567890");

        private readonly AsyncPackage package;

        private BoostAsioCommand(AsyncPackage package, OleMenuCommandService commandService)
        {
            this.package = package ?? throw new ArgumentNullException(nameof(package));
            commandService = commandService ?? throw new ArgumentNullException(nameof(commandService));

            var menuCommandID = new CommandID(CommandSet, CommandId);
            var menuItem = new MenuCommand(this.Execute, menuCommandID);
            commandService.AddCommand(menuItem);
        }

        public static BoostAsioCommand Instance { get; private set; }

        public static async Task InitializeAsync(AsyncPackage package)
        {
            OleMenuCommandService commandService = await package.GetServiceAsync(typeof(IMenuCommandService)) as OleMenuCommandService;
            Instance = new BoostAsioCommand(package, commandService);
        }

        private void Execute(object sender, EventArgs e)
        {
            System.Windows.Forms.MessageBox.Show("Boost Asio Extension Command Executed!");
        }
    }

## 📄 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.10)
project(AsioStubValidation LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Force Clang on Windows (if installed via LLVM)
set(CMAKE_CXX_COMPILER clang++)

# Find Boost
find_package(Boost REQUIRED COMPONENTS system)

include_directories(${Boost_INCLUDE_DIRS})

# Collect all .cpp files in asio_stubs
file(GLOB ASIO_STUBS "asio_stubs/*.cpp")

foreach(stub ${ASIO_STUBS})
    get_filename_component(stub_name ${stub} NAME_WE)
    add_executable(${stub_name} ${stub})
    target_link_libraries(${stub_name} ${Boost_LIBRARIES} pthread)
endforeach()
```

## 📄 build.bat (Automation Script for Windows)

```bat
@echo off
setlocal

echo === Building Asio stubs with Clang via CMake ===
if not exist build mkdir build
cd build
cmake .. -G "NMake Makefiles"
if errorlevel 1 (
    echo CMake configuration failed
    exit /b 1
)
nmake
if errorlevel 1 (
    echo Build failed
    exit /b 1
)
cd ..

echo === Building C# stubs with dotnet build ===
cd csharp_stubs
dotnet build
if errorlevel 1 (
    echo dotnet build failed
    exit /b 1
)
dotnet run
cd ..

echo === Validation complete ===
endlocal
```
