        {
            _context = context;
        }

        public Entry GetEntry(string header, string subheader, string name)
        {
            return _context.Entries.AsQueryable()
                .FirstOrDefault(e => e.Header == header && e.Subheader == subheader && e.Name == name);
        }

        public List<string> GetSubheaders(string header)
        {
            return _context.Entries.AsQueryable()
                .Where(e => e.Header == header)
                .Select(e => e.Subheader)
                .Distinct()
                .ToList();
        }

        public List<string> GetNames(string header, string subheader)
        {
            return _context.Entries.AsQueryable()
                .Where(e => e.Header == header && e.Subheader == subheader)
                .Select(e => e.Name)
                .Distinct()
                .ToList();
        }

        public List<Entry> GetAllEntries()
        {
            return _context.Entries.AsQueryable().ToList();
        }
    }

using System.IO;
using System.IO.Compression;
using System.Text;

namespace BoostAsioExtension
{
    public static class PlantUmlEncoder
    {
        public static string Encode(string text)
        {
            var bytes = Encoding.UTF8.GetBytes(text);
            using var output = new MemoryStream();
            using (var deflate = new DeflateStream(output, CompressionLevel.Optimal, true))
            {
                deflate.Write(bytes, 0, bytes.Length);
            }
            var compressed = output.ToArray();
            return Encode64(compressed);
        }

        private static string Encode64(byte[] data)
        {
            const string alphabet = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz-_";
            var sb = new StringBuilder();
            int i = 0;
            while (i < data.Length)
            {
                int b1 = data[i++] & 0xFF;
                int b2 = i < data.Length ? data[i++] & 0xFF : 0;
                int b3 = i < data.Length ? data[i++] & 0xFF : 0;

                int c1 = b1 >> 2;
                int c2 = ((b1 & 0x3) << 4) | (b2 >> 4);
                int c3 = ((b2 & 0xF) << 2) | (b3 >> 6);
                int c4 = b3 & 0x3F;

                sb.Append(alphabet[c1]);
                sb.Append(alphabet[c2]);
                sb.Append(alphabet[c3]);
                sb.Append(alphabet[c4]);
            }
            return sb.ToString();
        }
    }
